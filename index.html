extends Node2D # ---------------- PLAYER ---------------- var player_pos := Vector2(400, 300) var speed := 220 var health := 100 var flashlight := false var door_closed := false # ---------------- MONSTERS ---------------- var monsters: Array = [] var monster_speed := 95 # ---------------- SPAWN ---------------- var spawn_timer := 0.0 func _ready(): randomize() spawn_monster() spawn_monster() func _process(delta): handle_input(delta) move_monsters(delta) check_collisions() spawn_timer += delta if spawn_timer > 5.0: spawn_timer = 0 spawn_monster() queue_redraw() # ---------------- INPUT ---------------- func handle_input(delta): var dir := Vector2.ZERO if Input.is_action_pressed("move_up"): dir.y -= 1 if Input.is_action_pressed("move_down"): dir.y += 1 if Input.is_action_pressed("move_left"): dir.x -= 1 if Input.is_action_pressed("move_right"): dir.x += 1 if dir != Vector2.ZERO: dir = dir.normalized() player_pos += dir * speed * delta func _input(event): if event.is_action_pressed("flashlight"): flashlight = !flashlight if event.is_action_pressed("door"): door_closed = !door_closed # ---------------- MONSTERS ---------------- func spawn_monster(): var m = Vector2( randi_range(650, 900), randi_range(100, 500) ) monsters.append(m) func move_monsters(delta): for i in range(monsters.size()): var dir = (player_pos - monsters[i]).normalized() monsters[i] += dir * monster_speed * delta # ---------------- DAMAGE ---------------- func check_collisions(): for m in monsters: if m.distance_to(player_pos) < 35: if not door_closed: health -= 1 if health <= 0: get_tree().reload_current_scene() # ---------------- DRAW EVERYTHING ---------------- func _draw(): # room draw_rect(Rect2(0,0,1000,600), Color(0.08,0.08,0.12)) # hallway (danger zone) draw_rect(Rect2(500,0,500,600), Color(0.02,0.02,0.02)) # door if door_closed: draw_rect(Rect2(480,250,40,100), Color(0.6,0.2,0.1)) # player draw_circle(player_pos, 15, Color(0.3,0.7,1)) # monsters for m in monsters: draw_circle(m, 18, Color(1,0,0)) # flashlight glow if flashlight: draw_circle(player_pos, 120, Color(1,1,0.6,0.25))
