function drawSprites(){ let sprites = []; // pellets for(let p of pellets){ sprites.push({x:p.x,y:p.y,color:"yellow",size:6}); } // ghosts (FORCED visible overlay style) for(let g of ghosts){ sprites.push({x:g.x,y:g.y,color:g.color,size:14}); } // sort far → near sprites.sort((a,b)=>{ return Math.hypot(b.x-px,b.y-py)-Math.hypot(a.x-px,a.y-py); }); for(let s of sprites){ let dx = s.x - px; let dy = s.y - py; let dist = Math.hypot(dx,dy); let angle = Math.atan2(dy,dx) - pa; while(angle < -Math.PI) angle += Math.PI*2; while(angle > Math.PI) angle -= Math.PI*2; // wider FOV so nothing disappears const fov = Math.PI; if(Math.abs(angle) < fov){ let screenX = (angle/fov + 0.5) * canvas.width; // IMPORTANT FIX: no perspective scaling that hides ghosts let size = s.size * 10; ctx.fillStyle = s.color; ctx.beginPath(); ctx.arc(screenX, canvas.height/2, size, 0, Math.PI*2); ctx.fill(); // ghost “face glow” so you ALWAYS see them if(s.color !== "yellow"){ ctx.fillStyle = "white"; ctx.beginPath(); ctx.arc(screenX-3, canvas.height/2-2, 2, 0, Math.PI*2); ctx.arc(screenX+3, canvas.height/2-2, 2, 0, Math.PI*2); ctx.fill(); } } } }
